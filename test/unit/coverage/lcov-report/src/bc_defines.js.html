  errNoValidAddr: -5

export class PostcodeError extends Error {
  constructor (message, errCode) {
    super(message)
    if (Error.captureStackTrace) {
      Error.captureStackTrace(this, PostcodeError)
    }
 
    this.name = 'PostcodeError';
    this.errCode = errCode
  }
}
