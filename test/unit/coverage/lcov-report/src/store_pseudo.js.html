// import { reactive } from 'vue';
import { seedAddressData, seedBinCollnData } from './seed_pseudo.js';
 
export const store = {
  state: {
    addressData: seedAddressData,
    binCollnData: seedBinCollnData,
  },
  getPostAddresses (postcode) {
    // https://api.readingdev.com/rbc/getaddresses/{uk-postcode}
    // postcode should be ukpostcode without space
    console.log(postcode);
    // console.log(seedAddressData);
    let filteredAddresses = seedAddressData.filter((addressObj) => {
      return addressObj.postcode === postcode;
    });
    if (filteredAddresses) {
      console.log(filteredAddresses);
      return filteredAddresses;
    }
    return [];
  },
  findBinCollections (uprn) {
    // https://api.readingdev.com/api/collections/{uprn}
    console.log(uprn);
    let binCollnData = this.state.binCollnData.find((binCollnObj) => {
      return binCollnObj.uprn === uprn;
    });
    if (binCollnData) {
      console.log(binCollnData);
      return binCollnData;
    }
    return [];
  },
  getWeeklyBinCollectionSchedule (uprn) {
    return this.state.binCollnData.map((binCollnObj) => {
      return binCollnObj.date;
    });
  },
  setAddressData (postcode) {
    this.state.addressData.map((addressObj) => {
      addressObj.postcode === postcode ? addressObj.active = true : addressObj.active = false;
    });
  },
  submitEvent (eventDetails) {
    const activeDay = this.getActiveDay();
    activeDay.events.push({ "details": eventDetails, "edit": false });
  },
  editEvent (dayId, eventDetails) {
    this.resetEditOfAllEvents();
    const eventObj = this.getEventObj(dayId, eventDetails);
    eventObj.edit = true;
  },
  updateEvent (dayId, originalEventDetails, newEventDetails) {
    const eventObj = this.getEventObj(dayId, originalEventDetails);
    eventObj.details = newEventDetails;
    eventObj.edit = false;
  },
  deleteEvent (dayId, eventDetails) {
    const dayObj = this.state.data.find(
      day => day.id === dayId
    );
    const eventIndexToRemove = dayObj.events.findIndex(
      event => event.details === eventDetails
    );
    dayObj.events.splice(eventIndexToRemove, 1);
  },
  getEventObj (dayId, eventDetails) {
    const dayObj = this.state.data.find(
      day => day.id === dayId
    );
    return dayObj.events.find(
      event => event.details === eventDetails
    );
  },
  resetEditOfAllEvents () {
    this.state.data.map((dayObj) => {
      dayObj.events.map((event) => {
        event.edit = false;
      });
    });
  }
}
